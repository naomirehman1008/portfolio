75#ifndef htons
76#define htons(x) ((uint16_t)(((uint16_t)(x) & 0x00ff) << 8 | \
77 ((uint16_t)(x) & 0xff00) >> 8))
78#define ntohs(x) htons(x)
79#endif

81#ifndef htonl
82#define htonl(x) ((uint32_t)(((uint32_t)(x) & 0x000000ff) << 24 | \
83 ((uint32_t)(x) & 0x0000ff00) << 8 | \
84 ((uint32_t)(x) & 0x00ff0000) >> 8 | \
85 ((uint32_t)(x) & 0xff000000) >> 24))
86#define ntohl(x) htonl(x)
87#endif

103// Initialize the Pico I2C interface with needed parameters for
104// I2C controller <i2c_device> [i2c0 or i2c1, from hardware/i2c.h].
105// Address of VL53L1X sensor should be in <dev> (default value is 0x29)
106// Validates VL53L1X sensor is connected, and returns 0 on successful
107// initialization and validating the sensor. Returns -1 on failure.
